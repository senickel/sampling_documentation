  unlist() %>% 
  strsplit(";") %>% 
  do.call(rbind,.)

coord_degree <- coord_matrix[grep("°",coord_matrix[,1]),] %>% 
  apply(2,function(y) {
    y %>% 
      gsub("S","-",.,fixed=TRUE) %>% 
  gsub("[A-Z]","",.) %>%
  strsplit("[°\\.]") %>% 
  lapply(function(x) {
    x <- x %>% 
      gsub("[^0-9\\-]","",.) %>% 
      as.numeric()
    x[1] + x[2]/60 + x[3]/3600
  }) %>% 
      unlist()
})

  gsub("[A-Z]","",.) %>%
